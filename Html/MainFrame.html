 41      private BufferedImage resizeImage(BufferedImage originalImage, int width, int height, int type) throws IOException {  
 42         BufferedImage resizedImage = new BufferedImage(width, height, type);  
 43         Graphics2D g = resizedImage.createGraphics();  
 44         g.drawImage(originalImage, 0, 0, width, height, null);  
 45         g.dispose();  
 46         return resizedImage;  
 47     }
