309  // یک تابع برای انجام عمل زوم 
310     public void Zoom(int i) throws IOException {
311         Label.setIcon(new ImageIcon(resizeImage(nowImage,(nowImage.getWidth()+(i*100)),(nowImage.getHeight()+(i*75)), 5)));
312         nowImage=resizeImage(nowImage,(nowImage.getWidth()+(i*100)),(nowImage.getHeight()+(i*75)), 5);
313     }
314     //تابعی جهت تغییر اندازه عکس که از آن در تابع زوم استفاده میشود
315     //خروجی آن یک عکس تغییر اندازه داده شده است
316     public BufferedImage resizeImage(BufferedImage originalImage, int width, int height, int type) throws IOException {  
317         BufferedImage resizedImage = new BufferedImage(width, height, type);  
318         Graphics2D g = resizedImage.createGraphics();  
319         g.drawImage(originalImage, 0, 0, width, height, null);  
320         g.dispose();  
321         return resizedImage;  
322     }
